#Writing my query for the Gaia data of NGC 188. I suppose I don't need all of these parameters, but I have included them anyway. 
query = f"""
SELECT TOP 8000
source_id, ra, dec, phot_g_mean_mag, bp_rp, parallax, parallax_over_error
FROM gaiadr3.gaia_source_lite
WHERE DISTANCE( POINT({NGC188_ra}, {NGC188_dec}), POINT(ra, dec) ) < 1
AND parallax_over_error > 10
AND bp_rp IS NOT NULL
ORDER BY parallax DESC
"""

def abs_mag(my_gmag, my_distance):
    result = my_gmag - 5 * np.log10(my_distance / (10 * u.parsec)) * u.mag
    return result

query_table['abs_gmag'] = abs_mag(my_gmag = query_table['phot_g_mean_mag'], 
                                 my_distance = query_table['distance']
                               ) * u.mag
